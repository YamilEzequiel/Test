git version // vemos la version de Git

git init // Inisializando el proyecto

git status -s // estados de los archivos

git add index.html // Agregamos al area temporal los archivos indicados o git add . para todos los archivos

git commit -m "Mensaje" // Enviamos un commit con nuestro mensaje

git log --oneline // Vemos los commit realizados

git reset --hard N#Commit // volvemos en el commit del tiempo
